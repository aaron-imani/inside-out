#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

typedef struct {
    char *key;
    int value;
    struct Node *next;
} Node;

typedef struct {
    Node **table;
    size_t size;
} HashMap;

typedef struct {
    Node **array;
    size_t count;
} HashSet;

unsigned int hash(const char *str) {
    unsigned int hash = 5381;
    int c;
    while ((c = *str++))
        hash = ((hash << 5) + hash) + c;
    return hash;
}

HashMap* createHashMap(size_t size) {
    HashMap *map = (HashMap*)malloc(sizeof(HashMap));
    map->size = size;
    map->table = (Node**)calloc(size, sizeof(Node*));
    return map;
}

void freeHashMap(HashMap *map) {
    for (size_t i = 0; i < map->size; i++) {
        Node *node = map->table[i];
        while (node != NULL) {
            Node *temp = node;
            node = node->next;
            free(temp->key);
            free(temp);
        }
    }
    free(map->table);
    free(map);
}

bool containsKey(HashMap *map, const char *key) {
    unsigned int index = hash(key) % map->size;
    Node *node = map->table[index];
    while (node != NULL) {
        if (strcmp(node->key, key) == 0) return true;
        node = node->next;
    }
    return false;
}
