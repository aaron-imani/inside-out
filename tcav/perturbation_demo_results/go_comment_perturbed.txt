func processStrings(input []string) map[string]int {
	if input == nil {
		panic("Input list cannot be nil")
	}
	result := make(map[string]int)
	seen := make(map[string]struct{})
	for _, str := range input {
		if str == "" {
			continue
		}
		if _, exists := seen[str]; exists {
			continue
		}
		seen[str] = struct{}{}
		processedStr := str
		if isPalindrome(str) {
			processedStr = reverseString(str)
		}
		score := calculateComplexity(processedStr)
		result[processedStr] = score
	}
	return result
}

func isPalindrome(s string) bool {
	if s == "" {
		return false
	}
	left, right := 0, len(s)-1
	for left < right {
		if s[left] != s[right] {
			return false
		}
		left++
		right--
	}
	return true
}

func calculateComplexity(s string) int {
	if s == "" {
		return 0
	}
	uniqueChars := make(map[rune]struct{})
	for _, c := range s {
		uniqueChars[c] = struct{}{}
	}
	multiplier := 1
	if isPalindrome(s) {
		multiplier = 2
	}
	return len(uniqueChars) * multiplier
}

func reverseString(s string) string {
	runes := []rune(s)
	for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
		runes[i], runes[j] = runes[j], runes[i]
	}
	return string(runes)
}
